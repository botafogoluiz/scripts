#-*-coding: utf-8 -*-
from sikuli import *

def cadastro_empresas():
    wait(2)
    click()
    wait(1)
    click()
    type(Key.DOWN)
    type(Key.DOWN)
    # Push down keys.
    keyDown(Key.CTRL)
    keyDown(Key.SHIFT)
    type(Key.DOWN)
    # Release keys. 
    keyUp()
    wait(2)
    type("c",KEY_CTRL)
    wait(2)
    click()
    wait(2)
    click()
    wait(2)
    click(Pattern().similar(0.94).targetOffset(7,10))
    wait(2)
    type("v",KEY_CTRL)
    click()
    print"Cadastro Empresa Pronto"

def main():
    cadastro_empresas()

if __name__ == "__main__":
    main()
