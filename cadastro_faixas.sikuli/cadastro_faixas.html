from sikuli import *

def coluna_residencial():
    wait(3)
    click(Pattern().similar(0.82))
    wait(3)
    click(Pattern().similar(0.82))
    wait(2)
    type(Key.TAB)
    type(Key.TAB)
    wait(2)
    type(Key.DOWN)
    wait(2)
    # Push down keys.
    keyDown(Key.CTRL)
    keyDown(Key.SHIFT)
    type(Key.DOWN)
    type(Key.DOWN)
    type(Key.DOWN)
    type(Key.DOWN)
    type(Key.DOWN)
    type(Key.DOWN)
    type(Key.DOWN)
    # Release keys. 
    keyUp()
    wait(2)
    type("c",KEY_CTRL)
    click()
    wait(2)
    click()
    wait(2)
    click()
    type(Key.TAB)
    wait(2)
    type(Key.TAB)
    wait(2)
    type(Key.DOWN)
    type("v",KEY_CTRL)
    wait(2)
    click()
    type(Key.UP)
    print("Residencial pronto")
