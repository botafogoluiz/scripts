from sikuli import *

def substituicao_hds():
    wait(2)
    click()
    wait(2)
    click()
    type(Key.DOWN)
    type(Key.DOWN)
    type(Key.RIGHT)
    type(Key.RIGHT)
    # Copia da primeira coluna
    # Push down keys.
    keyDown(Key.CTRL)
    keyDown(Key.SHIFT)
    for i in range (0,23):
        type(Key.DOWN)
    # Release keys. 
    keyUp()
    wait(2)
    type("c",KEY_CTRL)
    wait(2)
    click()
    wait(1)
    hover()
    click()
    wait(2)
    click(Pattern().targetOffset(288,14))
    wait(2)
    type("v",KEY_CTRL)
    wait(5)
    click()
    type(Key.RIGHT)
    for i in range (0,147):
        type(Key.DOWN)
    # Push down keys.
    keyDown(Key.CTRL)
    keyDown(Key.SHIFT)
    for i in range (0,24):
        type(Key.DOWN)
    for i in range (0,3):
        type(Key.RIGHT)
    # Release keys. 
    keyUp()
    wait(2)
    type("c",KEY_CTRL)
    wait(2)
    click()
    wait(2)
    type(Key.RIGHT)
    for i in range (0,148):
        type(Key.DOWN)
        wait(0.5)
    wait(2)
    type("v",KEY_CTRL)
    wait(3)
    print "Substituicao de HDS copiado"
    click()

def main():
    substituicao_hds()

if __name__ == "__main__":
    main()
